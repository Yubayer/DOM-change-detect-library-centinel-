DOMSentinel

Production-grade DOM mutation detection system

v2.1 Zero dependency ~2KB minified MutationObserver

DOMSentinel একটা lightweight JavaScript library যেটা দিয়ে DOM-এর যেকোনো পরিবর্তন detect করা যায় এবং সেই পরিবর্তনের উপর ভিত্তি করে custom event fire করা যায়।

একটাই MutationObserver পুরো document watch করে। তুমি যতগুলো খুশি watcher register করতে পারো — প্রতিটা নিজের মতো কাজ করবে, একটা আরেকটাকে প্রভাবিত করবে না।

কীভাবে কাজ করে:
DOM বদলায় → MutationObserver ধরে → প্রতিটা watcher match check করে → filter pass হলে → debounce → CustomEvent dispatch + callback call

Library Code

এই পুরো code টা তোমার project-এ copy করো। কোনো dependency নেই।

/* ============================================================
 * DOMSentinel v2.1 — Production-grade DOM mutation detection
 * Zero dependency | ~2KB | MutationObserver-based
 * ============================================================ */

class DOMSentinel {

  /* ── private state ── */
  #observer       = null;   // single shared MutationObserver
  #watchers       = [];     // active WatchConfig objects
  #debounceTimers = new Map(); // watchId → setTimeout handle
  #triggerCounts  = new Map(); // watchId → fire count
  #paused         = false;   // skip all processing when true
  #idCounter      = 0;       // monotonic watcher id
  #debug          = false;   // verbose logging

  /**
   * @param {{ debug?: boolean }} [options]
   */
  constructor({ debug = false } = {}) {
    this.#debug = debug;
    this.#observer = new MutationObserver(this.#onMutations.bind(this));
    this.#observer.observe(document.documentElement, {
      childList:             true,
      subtree:               true,
      attributes:            true,
      characterData:         true,
      attributeOldValue:     true,
      characterDataOldValue: true,
    });
    this.#log('initialized');
  }

  /* ── mutation pipeline ── */

  #onMutations(mutations) {
    if (this.#paused) return;
    for (const mutation of mutations) {
      for (const watcher of this.#watchers) {
        try {
          const matched = this.#resolveMatch(mutation, watcher);
          if (matched) this.#handleMatch(matched, mutation, watcher);
        } catch (err) {
          console.error(`[DOMSentinel] watcher #${watcher.id} threw:`, err);
        }
      }
    }
  }

  #resolveMatch(mutation, watcher) {
    const { selector, relation, relationSelector } = watcher;
    switch (relation) {
      case 'target': return this.#matchTarget(mutation, selector, relationSelector);
      case 'parent': return this.#matchParent(mutation, selector, relationSelector);
      case 'child':  return this.#matchChild(mutation, selector, relationSelector);
      default:       return null;
    }
  }

  #matchTarget(mutation, selector, relationSelector) {
    if (mutation.type === 'childList') {
      if (
        relationSelector &&
        mutation.target.nodeType === Node.ELEMENT_NODE &&
        !mutation.target.closest(relationSelector)
      ) return null;

      for (const node of [...mutation.addedNodes, ...mutation.removedNodes]) {
        if (node.nodeType !== Node.ELEMENT_NODE) continue;
        if (node.matches(selector)) return node;
        const inner = node.querySelector?.(selector);
        if (inner) return inner;
      }
    }
    if (mutation.type === 'attributes' || mutation.type === 'characterData') {
      const el = mutation.target.nodeType === Node.ELEMENT_NODE
        ? mutation.target
        : mutation.target.parentElement;
      if (el?.matches(selector)) return el;
    }
    return null;
  }

  #matchParent(mutation, selector, relationSelector) {
    if (mutation.type !== 'childList') return null;
    const parent = mutation.target;
    if (parent.nodeType !== Node.ELEMENT_NODE) return null;
    if (relationSelector && !parent.matches(relationSelector)) return null;
    for (const node of [...mutation.addedNodes, ...mutation.removedNodes]) {
      if (node.nodeType !== Node.ELEMENT_NODE) continue;
      if (node.matches(selector) || node.querySelector?.(selector)) return parent;
    }
    return null;
  }

  #matchChild(mutation, selector, relationSelector) {
    const origin = mutation.target.nodeType === Node.ELEMENT_NODE
      ? mutation.target
      : mutation.target.parentElement;
    const target = origin?.closest(selector) ?? null;
    if (!target) return null;
    if (relationSelector && !origin.closest(relationSelector)) return null;
    return target;
  }

  #handleMatch(element, mutation, watcher) {
    const detail = this.#buildDetail(element, mutation, watcher);
    if (watcher.filter && !watcher.filter(element, detail)) return;
    if (watcher.debounce > 0) {
      clearTimeout(this.#debounceTimers.get(watcher.id));
      this.#debounceTimers.set(watcher.id, setTimeout(() => {
        this.#debounceTimers.delete(watcher.id);
        this.#fire(element, detail, watcher);
      }, watcher.debounce));
    } else {
      this.#fire(element, detail, watcher);
    }
  }

  #fire(element, detail, watcher) {
    const count = (this.#triggerCounts.get(watcher.id) ?? 0) + 1;
    this.#triggerCounts.set(watcher.id, count);
    this.#log(`watcher #${watcher.id} fired ×${count}`);
    try {
      document.dispatchEvent(new CustomEvent(watcher.eventName, { bubbles: false, detail }));
    } catch (err) {
      console.error(`[DOMSentinel] dispatch failed:`, err);
    }
    if (typeof watcher.callback === 'function') {
      try { watcher.callback(detail); } catch (err) {
        console.error(`[DOMSentinel] callback error:`, err);
      }
    }
    if (watcher.once || (watcher.maxTriggers > 0 && count >= watcher.maxTriggers)) {
      this.unwatch(watcher.id);
    }
  }

  #buildDetail(element, mutation, watcher) {
    return {
      element,
      watcherId:        watcher.id,
      relation:         watcher.relation,
      relationSelector: watcher.relationSelector,
      mutationType:     mutation.type,
      added:            [...(mutation.addedNodes   ?? [])],
      removed:          [...(mutation.removedNodes ?? [])],
      attributeName:    mutation.attributeName  ?? null,
      oldValue:         mutation.oldValue       ?? null,
      timestamp:        Date.now(),
    };
  }

  #log(...args) {
    if (this.#debug) console.log('[DOMSentinel]', ...args);
  }

  /* ── public API ── */

  watch({
    selector,
    relation         = 'target',
    relationSelector = null,
    eventName,
    callback         = null,
    filter           = null,
    debounce         = 0,
    once             = false,
    maxTriggers      = 0,
  } = {}) {
    if (!selector || typeof selector !== 'string')
      throw new TypeError('[DOMSentinel] "selector" must be a non-empty string.');
    if (!eventName || typeof eventName !== 'string')
      throw new TypeError('[DOMSentinel] "eventName" must be a non-empty string.');
    if (!['target', 'parent', 'child'].includes(relation))
      throw new TypeError('[DOMSentinel] "relation" must be: target | parent | child');
    if (filter   !== null && typeof filter   !== 'function')
      throw new TypeError('[DOMSentinel] "filter" must be a function or null.');
    if (callback !== null && typeof callback !== 'function')
      throw new TypeError('[DOMSentinel] "callback" must be a function or null.');

    const id = ++this.#idCounter;
    this.#watchers.push({
      id, selector, relation, relationSelector,
      eventName, callback, filter,
      debounce:    Math.max(0, Number(debounce)    || 0),
      once:        Boolean(once),
      maxTriggers: Math.max(0, Number(maxTriggers) || 0),
    });
    this.#log(`watcher #${id} registered`, selector);
    return id;
  }

  unwatch(watchId) {
    clearTimeout(this.#debounceTimers.get(watchId));
    this.#debounceTimers.delete(watchId);
    this.#triggerCounts.delete(watchId);
    this.#watchers = this.#watchers.filter(w => w.id !== watchId);
  }

  pause()   { this.#paused = true;  this.#log('paused');  }
  resume()  { this.#paused = false; this.#log('resumed'); }
  get isPaused() { return this.#paused; }

  listWatchers() {
    return this.#watchers.map(w => ({
      id: w.id, selector: w.selector, relation: w.relation,
      relationSelector: w.relationSelector, eventName: w.eventName,
      debounce: w.debounce, once: w.once, maxTriggers: w.maxTriggers,
      triggerCount: this.#triggerCounts.get(w.id) ?? 0,
    }));
  }

  destroy() {
    this.#observer.disconnect();
    this.#debounceTimers.forEach(clearTimeout);
    this.#debounceTimers.clear();
    this.#triggerCounts.clear();
    this.#watchers = [];
    this.#log('destroyed');
  }
}

ব্যবহার শুরু

একটা instance বানাও, তারপর যত খুশি watcher register করো।

const sentinel = new DOMSentinel(); // production
const sentinel = new DOMSentinel({ debug: true }); // development

detail object — প্রতিটা event আর callback-এ এই data পাবে:

// e.detail বা callback-এর parameter
{
  element:          Element,   // matched DOM element
  watcherId:        Number,    // কোন watcher fire করেছে
  relation:         String,    // 'target' | 'parent' | 'child'
  relationSelector: String,    // narrowing selector (বা null)
  mutationType:     String,    // 'childList' | 'attributes' | 'characterData'
  added:            Node[],    // নতুন nodes
  removed:          Node[],    // সরানো nodes
  attributeName:    String,    // কোন attribute বদলেছে (বা null)
  oldValue:         String,    // আগের value (বা null)
  timestamp:        Number,    // Date.now()
}

watch()

নতুন watcher register করো। watchId return করে — পরে unwatch()-এ লাগবে।

ParameterTypeDefaultRequiredকী করে
selectorstring—requiredকোন element watch করবে (CSS selector)
relationstringtargetoptionalকোন ধরনের change — target / parent / child
relationSelectorstringnulloptionalrelation আরো narrow করে
eventNamestring—requiredmatch হলে কোন CustomEvent fire হবে
callbackfunctionnulloptionalinline function — event-এর পাশাপাশি চলে
filterfunctionnulloptionalmatch drop বা allow করার predicate
debouncenumber (ms)0optionalrapid mutations batch করে
oncebooleanfalseoptionalএকবার fire করে auto-remove
maxTriggersnumber0optionalN বার fire-এর পর auto-remove (0 = unlimited)

unwatch(watchId)

নির্দিষ্ট watcher সরিয়ে দাও। Pending debounce timer-ও cancel হয়ে যায়।

const id = sentinel.watch({ selector: '.card', eventName: 'card:x' });
sentinel.unwatch(id); // সরিয়ে দাও

pause() / resume()

নিজের code-এর DOM write-এ sentinel react করবে না — তার আগে pause(), পরে resume()।

pause চলাকালীন যে mutations হয় সেগুলো permanently lost — queue হয় না।
function replaceCards(html) {
  sentinel.pause();
  grid.innerHTML = html;   // আমাদের নিজের write
  sentinel.resume();
}

// বর্তমান state check করতে
if (sentinel.isPaused) { /* ... */ }

listWatchers()

সব active watcher-এর snapshot — debugging-এ কাজে আসে।

console.table(sentinel.listWatchers());
// id | selector | relation | eventName | debounce | triggerCount ...

destroy()

Observer disconnect করো, সব watcher clear করো, সব timer cancel করো। Page unload বা SPA route change-এ অবশ্যই call করো।

window.addEventListener('beforeunload', () => sentinel.destroy());

relation

তুমি যে element watch করতে চাও তার সাথে mutation-এর সম্পর্ক কী।

target

Element নিজেই DOM-এ add/remove হয়েছে, অথবা তার নিজের attribute/text বদলেছে।

sentinel.watch({ selector: '.card-product', relation: 'target', eventName: 'card:added' });

parent

Element-এর parent-এ child add/remove হয়েছে। Matched element হিসেবে parent-কে return করে।

// .product-grid এর ভেতরে card ঢুকলে → grid টাকে return করবে
sentinel.watch({ selector: '.card-product', relation: 'parent', eventName: 'grid:changed' });

child

Element-এর ভেতরে কিছু বদলেছে। closest() দিয়ে উপরে উঠে enclosing element খুঁজে বের করে।

// .card-product এর ভেতরে যেকোনো কিছু বদলালে → card-কে return করবে
sentinel.watch({ selector: '.card-product', relation: 'child', eventName: 'card:changed' });

relationSelector

relation-কে আরো specific করে। তিনটা relation-এই কাজ করে।

// target: শুধু .product-grid এর ভেতরে card add হলে fire করবে
sentinel.watch({
  selector:         '.card-product',
  relation:         'target',
  relationSelector: '.product-grid',  // ← শুধু এই container
  eventName:        'card:in-grid',
});

// parent: parent অবশ্যই .product-grid হতে হবে
sentinel.watch({
  selector:         '.card-product',
  relation:         'parent',
  relationSelector: '.product-grid',  // ← শুধু এই parent
  eventName:        'grid:updated',
});

// child: mutation অবশ্যই .card-product__price এর ভেতরে হতে হবে
sentinel.watch({
  selector:         '.card-product',
  relation:         'child',
  relationSelector: '.card-product__price',  // ← শুধু এই subtree
  eventName:        'card:price-changed',
});

filter

Relation match হওয়ার পরে এই function চলে। false return করলে সেই match silently drop হয়।

sentinel.watch({
  selector:  '.card-product',
  relation:  'target',
  eventName: 'card:unloaded',

  // শুধু তখনই fire করো যখন review load হয়নি
  filter: (element, detail) => {
    return element.dataset.reviewLoaded !== 'true';
  },
});
filter-এ দুটো parameter আসে: element (matched DOM element) আর detail (পুরো mutation info)।

callback

EventListener আলাদা করে লেখা ছাড়াই inline function দিতে পারো। Event আর callback একসাথে fire হয় — একটা আরেকটাকে replace করে না।

sentinel.watch({
  selector:  '.card-product',
  eventName: 'card:added',

  // inline কাজ: review inject করো
  callback: ({ element: card }) => {
    fetchReview(card.dataset.productId);
  },
});

// আলাদা concern: analytics
document.addEventListener('card:added', ({ detail }) => {
  analytics.track('card_impression', { id: detail.element.dataset.productId });
});

debounce

AJAX filter করলে ২০-৩০টা card একসাথে add হয়। প্রতিটার জন্য আলাদা API call না করে debounce দাও — burst শেষ হলে একবার fire করবে।

sentinel.watch({
  selector:  '.card-product',
  relation:  'target',
  eventName: 'card:settled',
  debounce:  100,  // ১০০ms চুপ থাকলে তবেই fire করবে
});
card add → card add → card add → ... 100ms চুপ → 🔥 একবার fire

once

প্রথমবার fire করার পরে watcher নিজেই সরে যায়।

sentinel.watch({
  selector:  '.card-product',
  relation:  'target',
  eventName: 'card:first-ever',
  once:      true,
  callback:  () => initTooltips(), // প্রথম কার্ড এলেই init, তারপর বন্ধ
});

maxTriggers

N বার fire হওয়ার পরে watcher auto-remove হয়ে যায়। once: true আর maxTriggers একসাথে দিলে once জেতে।

sentinel.watch({
  selector:    '.card-product',
  eventName:   'card:impression',
  maxTriggers: 5,  // ৫বার fire হলে auto-remove
  callback:    ({ element }) => analytics.track('impression', { id: element.dataset.productId }),
});
